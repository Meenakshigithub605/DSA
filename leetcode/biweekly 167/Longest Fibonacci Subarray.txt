class Solution {
public:
    int longestSubarray(vector<int>& nums) {
        int len=2,curr=2;
        for(int i=2;i<nums.size();i++){
            if(nums[i]==1LL*(nums[i-1]+nums[i-2])) curr+=1;
            else curr=2;
            len=max(len,curr);
        }
        return len;
    }
