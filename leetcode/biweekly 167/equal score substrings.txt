class Solution {
public:
    bool scoreBalance(string s) {
        vector<int>ans;
        int sum=0;
        for(int i=0;i<s.length();i++){
            sum+=s[i]-'a'+1;
            ans.push_back(sum);
        }
        if(sum%2==1) return false;
        for(int i=0;i<s.length();i++){
            if(ans[i]==sum/2) return true;
        }
        return false;
    }
