class Solution {
public:
    int longestSubsequence(vector<int>& nums) {
        int xor1=0;
        bool nonzero=false;
        for(int i=0;i<nums.size();i++){
            xor1^=nums[i];
            if(nums[i]!=0) nonzero=true;
        }
        if(xor1!=0) return nums.size();
        else{
            if(nonzero) return nums.size()-1;
            else return 0;
        }
    }
};
